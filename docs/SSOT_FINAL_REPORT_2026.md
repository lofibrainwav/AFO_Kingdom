def bracket_tax_py(income, thresholds, rates):
    tax = 0.0
    prev = 0.0
    for cap, r in zip(thresholds, rates[:-1]):
        if income <= cap:
            return tax + (income - prev) * r
        tax += (cap - prev) * r
        prev = cap
    return tax + (income - prev) * rates[-1]

@njit(cache=True)

        r = rates[i]

        tax += (cap - prev) * r
        prev = cap
